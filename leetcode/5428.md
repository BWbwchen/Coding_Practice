# 5428. Shuffle the Array


```c++=
class Solution {
public:
    vector<int> shuffle(vector<int>& nums, int n) {
        vector<int > ans (nums.size(), 0);
        for (int i = 0; i < nums.size(); ++i) {
            if (i %2 == 0) {
                ans[i] = nums[i/2];
            } else {
                ans[i] = nums[n+i/2];
            }
        }
        return ans;
    }
};
```
